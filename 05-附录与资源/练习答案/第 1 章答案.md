## 练习 2：时间问候

```java
package com.langgraph4j.exercises;

import org.bsc.langgraph4j.StateGraph;
import org.bsc.langgraph4j.CompiledGraph;
import org.bsc.langgraph4j.state.AgentState;
import org.bsc.langgraph4j.state.Channel;

import java.util.Map;
import java.util.HashMap;

public class Exercise2_TimeGreeting {

    public static void main(String[] args) {
        // 1. 定义状态通道
        Map<String, Channel<?>> channels = new HashMap<>();
        channels.put("name", Channel.of());
        channels.put("greeting", Channel.of());

        // 2. 创建状态图
        StateGraph graph = new StateGraph(channels);

        // 3. 添加带时间问候的节点
        graph.addNode("greet", state -> {
            int hour = java.time.LocalTime.now().getHour();
            String timeGreeting;
            if (hour >= 5 && hour < 12) {
                timeGreeting = "早上好";
            } else if (hour >= 12 && hour < 18) {
                timeGreeting = "下午好";
            } else {
                timeGreeting = "晚上好";
            }
            String name = state.value("name").orElse("World");
            String greeting = timeGreeting + ", " + name + "!";
            state.data().put("greeting", greeting);
            System.out.println("[DEBUG] 生成问候语：" + greeting);
            return state;
        });

        // 4. 添加边
        graph.addEdge("__start__", "greet");
        graph.addEdge("greet", "__end__");

        // 5. 编译
        CompiledGraph compiled = graph.compile();

        // 6. 执行
        Map<String, Object> inputData = new HashMap<>();
        inputData.put("name", "张三");
        AgentState result = (AgentState) compiled.run(inputData);

        // 7. 输出结果
        System.out.println(result.value("greeting").orElse("No greeting"));
    }
}
```

## 练习 3：三节点工作流

```java
package com.langgraph4j.exercises;

import org.bsc.langgraph4j.StateGraph;
import org.bsc.langgraph4j.CompiledGraph;
import org.bsc.langgraph4j.state.AgentState;
import org.bsc.langgraph4j.state.Channel;

import java.util.Map;
import java.util.HashMap;

/**
 * 三节点工作流：extract → transform → output
 */
public class Exercise3_ThreeNodes {

    public static void main(String[] args) {
        // 1. 定义状态通道
        Map<String, Channel<?>> channels = new HashMap<>();
        channels.put("rawInput", Channel.of());
        channels.put("extractedName", Channel.of());
        channels.put("upperName", Channel.of());
        channels.put("finalOutput", Channel.of());

        // 2. 创建状态图
        StateGraph graph = new StateGraph(channels);

        // 3. extract 节点：从输入中提取用户名
        graph.addNode("extract", state -> {
            String rawInput = state.value("rawInput").orElse("");
            // 简单提取：假设名字在 "hello, " 之后
            String name = rawInput.toLowerCase().replace("hello,", "")
                                   .replace("hi,", "").trim();
            state.data().put("extractedName", name);
            System.out.println("[extract] 提取的名字：" + name);
            return state;
        });

        // 4. transform 节点：将名字转换为大写
        graph.addNode("transform", state -> {
            String name = state.value("extractedName").orElse("");
            String upperName = name.toUpperCase();
            state.data().put("upperName", upperName);
            System.out.println("[transform] 大写名字：" + upperName);
            return state;
        });

        // 5. output 节点：生成最终输出
        graph.addNode("output", state -> {
            String upperName = state.value("upperName").orElse("");
            String finalOutput = "处理完成：" + upperName;
            state.data().put("finalOutput", finalOutput);
            System.out.println("[output] 最终输出：" + finalOutput);
            return state;
        });

        // 6. 添加边
        graph.addEdge("__start__", "extract");
        graph.addEdge("extract", "transform");
        graph.addEdge("transform", "output");
        graph.addEdge("output", "__end__");

        // 7. 编译
        CompiledGraph compiled = graph.compile();

        // 8. 执行
        Map<String, Object> inputData = new HashMap<>();
        inputData.put("rawInput", "hello, alice");
        AgentState result = (AgentState) compiled.run(inputData);

        // 9. 输出结果
        System.out.println("=== 最终结果 ===");
        System.out.println(result.value("finalOutput").orElse("No output"));
        // 预期输出：处理完成：ALICE
    }
}
```

## 练习 4：多条件分支

```java
package com.langgraph4j.exercises;

import org.bsc.langgraph4j.StateGraph;
import org.bsc.langgraph4j.CompiledGraph;
import org.bsc.langgraph4j.state.AgentState;
import org.bsc.langgraph4j.state.Channel;

import java.util.Map;
import java.util.HashMap;

/**
 * 三分支条件问候：VIP / 会员 / 普通用户
 */
public class Exercise4_MultiBranch {

    // 路由函数
    public static String route(AgentState state) {
        String userType = state.value("userType").orElse("normal");
        return switch (userType) {
            case "vip" -> "vipGreet";
            case "member" -> "memberGreet";
            default -> "normalGreet";
        };
    }

    public static void main(String[] args) {
        // 1. 定义状态通道
        Map<String, Channel<?>> channels = new HashMap<>();
        channels.put("userType", Channel.of());
        channels.put("greeting", Channel.of());

        // 2. 创建状态图
        StateGraph graph = new StateGraph(channels);

        // 3. VIP 问候节点
        graph.addNode("vipGreet", state -> {
            state.data().put("greeting", "尊敬的 VIP 用户，欢迎您的光临！");
            return state;
        });

        // 4. 会员问候节点
        graph.addNode("memberGreet", state -> {
            state.data().put("greeting", "亲爱的会员，欢迎光临！");
            return state;
        });

        // 5. 普通用户问候节点
        graph.addNode("normalGreet", state -> {
            state.data().put("greeting", "您好，欢迎光临！");
            return state;
        });

        // 6. 添加条件边
        graph.addConditionalEdges("__start__", Exercise4_MultiBranch::route,
            Map.of(
                "vipGreet", "vipGreet",
                "memberGreet", "memberGreet",
                "normalGreet", "normalGreet"
            ));

        // 7. 添加结束边
        graph.addEdge("vipGreet", "__end__");
        graph.addEdge("memberGreet", "__end__");
        graph.addEdge("normalGreet", "__end__");

        // 8. 编译
        CompiledGraph compiled = graph.compile();

        // 9. 测试不同类型用户
        testUser(compiled, "vip", "=== VIP 用户测试 ===");
        testUser(compiled, "member", "=== 会员用户测试 ===");
        testUser(compiled, "normal", "=== 普通用户测试 ===");
        testUser(compiled, null, "=== 默认用户测试 ===");
    }

    private static void testUser(CompiledGraph compiled, String userType, String label) {
        System.out.println(label);
        Map<String, Object> input = new HashMap<>();
        input.put("userType", userType);
        AgentState result = (AgentState) compiled.run(input);
        System.out.println(result.value("greeting").orElse(""));
        System.out.println();
    }
}
```

**运行输出**：
```
=== VIP 用户测试 ===
尊敬的 VIP 用户，欢迎您的光临！

=== 会员用户测试 ===
亲爱的会员，欢迎光临！

=== 普通用户测试 ===
您好，欢迎光临！

=== 默认用户测试 ===
您好，欢迎光临！
```

## 练习 5：执行轨迹日志

```java
package com.langgraph4j.exercises;

import org.bsc.langgraph4j.StateGraph;
import org.bsc.langgraph4j.CompiledGraph;
import org.bsc.langgraph4j.state.AgentState;
import org.bsc.langgraph4j.state.Channel;

import java.util.Map;
import java.util.HashMap;
import java.util.concurrent.atomic.AtomicInteger;

/**
 * 带执行轨迹日志的工作流
 */
public class Exercise5_ExecutionTrace {

    private static final AtomicInteger stepCounter = new AtomicInteger(0);

    public static void main(String[] args) {
        System.out.println("========== 工作流执行开始 ==========");
        stepCounter.set(0);

        // 1. 定义状态通道
        Map<String, Channel<?>> channels = new HashMap<>();
        channels.put("name", Channel.of());
        channels.put("greeting", Channel.of());
        channels.put("traceLog", Channel.of());

        // 2. 创建状态图
        StateGraph graph = new StateGraph(channels);

        // 3. 添加带轨迹的节点
        graph.addNode("greet", state -> {
            int step = stepCounter.incrementAndGet();
            String trace = "[TRACE] 步骤 " + step + ": 进入 greet 节点\n";

            String name = state.value("name").orElse("Unknown");
            trace += "[TRACE] 步骤 " + step + ": 读取 name = " + name + "\n";

            String greeting = "Hello, " + name + "! Welcome to LangGraph4J!";
            state.data().put("greeting", greeting);
            trace += "[TRACE] 步骤 " + step + ": 生成 greeting = " + greeting + "\n";

            // 累积轨迹日志
            String existingTrace = state.value("traceLog").orElse("");
            state.data().put("traceLog", existingTrace + trace);

            return state;
        });

        // 4. 添加边
        graph.addEdge("__start__", "greet");
        graph.addEdge("greet", "__end__");

        // 5. 编译
        CompiledGraph compiled = graph.compile();

        // 6. 执行
        Map<String, Object> inputData = new HashMap<>();
        inputData.put("name", "Developer");
        AgentState result = (AgentState) compiled.run(inputData);

        // 7. 输出结果和轨迹
        System.out.println("========== 工作流执行结束 ==========\n");
        System.out.println("最终输出：" + result.value("greeting").orElse(""));
        System.out.println("\n执行轨迹:");
        System.out.println(result.value("traceLog").orElse("No trace"));
    }
}
```

**运行输出**：
```
========== 工作流执行开始 ==========
========== 工作流执行结束 ==========

最终输出：Hello, Developer! Welcome to LangGraph4J!

执行轨迹:
[TRACE] 步骤 1: 进入 greet 节点
[TRACE] 步骤 1: 读取 name = Developer
[TRACE] 步骤 1: 生成 greeting = Hello, Developer! Welcome to LangGraph4J!
```

## 思考题解答

### 1. 如果节点执行失败，会发生什么？

当节点执行失败时：
- 默认情况下，`CompiledGraph.run()` 会抛出异常
- 异常类型通常是 `RuntimeException` 或节点中抛出的具体异常
- 工作流会立即停止，不会继续执行后续节点

**处理方式**：
```java
try {
    AgentState result = (AgentState) compiled.run(inputData);
} catch (Exception e) {
    System.err.println("工作流执行失败：" + e.getMessage());
    // 可以进行错误恢复或记录日志
}
```

### 2. 如何记录完整的执行历史？

有几种方式：

**方式一：在状态中累积日志**（如练习 5 所示）

**方式二：使用 LangGraph4J 的 Hook 机制**
```java
graph.addBeforeCallNodeHook(state -> {
    System.out.println("[HOOK] 即将执行节点");
    return state;
});

graph.addAfterCallNodeHook(state -> {
    System.out.println("[HOOK] 节点执行完成");
    return state;
});
```

**方式三：使用外部日志框架**
```java
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

private static final Logger logger = LoggerFactory.getLogger(MyWorkflow.class);

graph.addNode("greet", state -> {
    logger.info("执行 greet 节点，输入：{}", state.value("name").orElse(""));
    // ...
});
```
